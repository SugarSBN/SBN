# 求 a^b mod m (b很大)

```c++
scanf("%lld%lld",&a,&m);
	ll p = phi(m);
	char c;
	bool bigger = 0;
	while(!isdigit(c))	c = getchar();
	ll b = 0;
	for(;isdigit(c);c = getchar()){
		b = b * 10 + c - '0';
		if (b >= p)	{
			bigger = 1;
			b %= p;
		}
	}
	if (bigger) b += p;
	ll ans = qpow(a,b,m);
	printf("%lld\n",ans);
```
