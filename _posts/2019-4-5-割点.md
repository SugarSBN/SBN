void tarjan(int u){
	dfn[u] = low[u] = ++clo;
	int ch = 0;
	for (int i = head[u];i;i = E[i].nxt){
		int v = E[i].y;
		if (!dfn[v]){
			tarjan(v);
			low[u] = min(low[u],low[v]);
			if (low[v] >= dfn[u] && u != root)	cut[u] = 1;
			if (u == root)	ch++;
		}
		low[u] = min(low[u],dfn[v]);
	}
	if (u == root && ch >= 2)	cut[u] = 1;
}
